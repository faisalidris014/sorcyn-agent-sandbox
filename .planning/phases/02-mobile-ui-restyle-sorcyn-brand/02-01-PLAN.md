    // Source: FRONTEND_RESTYLING_PLAN.md "Spring-based page route transitions"

    //   _kMass = 1.0
    //   _kTransitionDuration = Duration(milliseconds: 420)  // spring settles ~400ms at these params

    // Public API:
    //   CustomTransitionPage<T> springPage<T>({required LocalKey key, required Widget child, ...})
    //   class _SpringCurve extends Curve { ... } (private)

    // The transitionsBuilder applies a slide+fade combo:

    //   - FadeTransition: Tween<double>(begin: 0.0, end: 1.0) animated by springCurvedAnimation
    // No scale (avoid jank with bottom nav shell). Slide is subtle (4% of height) to avoid feeling like a heavy push transition.

    ```dart
    // Reusable tap-scale wrapper applying the locked Sorcyn `active:scale-[0.97]` token
    // (per ROADMAP Phase 2 SC #4 + FRONTEND_RESTYLING_PLAN.md "Active states").
    //
    // Use this for any tappable widget that is NOT already gradient_button / gradient_fab /
    // social_auth_button (those embed the same animation internally). Examples: filter chips,
    // post cards, conversation tiles, settings rows, avatar buttons.

    class TapScale extends StatefulWidget {
      final Widget child;
      final VoidCallback? onTap;
      final Duration duration; // default const Duration(milliseconds: 100) — matches gradient_button
      final double pressedScale; // default 0.97 — locked token
      final HitTestBehavior behavior; // default HitTestBehavior.opaque

      const TapScale({
        super.key,
        required this.child,
        this.onTap,
        this.duration = const Duration(milliseconds: 100),
        this.pressedScale = 0.97,
        this.behavior = HitTestBehavior.opaque,
      });

      @override
      State<TapScale> createState() => _TapScaleState();
    }

    class _TapScaleState extends State<TapScale> {
      bool _pressed = false;

      @override
      Widget build(BuildContext context) {
        final disabled = widget.onTap == null;
        return GestureDetector(
          behavior: widget.behavior,
          onTapDown: disabled ? null : (_) => setState(() => _pressed = true),
          onTapUp: disabled
              ? null
              : (_) {
                  setState(() => _pressed = false);
                  widget.onTap?.call();
                },
          onTapCancel: disabled ? null : () => setState(() => _pressed = false),
          child: AnimatedScale(
            scale: _pressed ? widget.pressedScale : 1.0,
            duration: widget.duration,
            child: widget.child,
          ),
        );
      }
    }
    ```
