```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';
import { SharedArray } from 'k6/data';

export const options = {
  scenarios: {
    transaction_loop: {
      executor: 'ramping-vus',
      startVUs: 0,
      stages: [
        { duration: '5m', target: 1000 },
        { duration: '15m', target: 1000 },
        { duration: '2m', target: 0 },
      ],
    },
  },
  thresholds: {
    'http_req_duration{type:api}': ['p(95)<500'],
    'http_req_duration{type:search}': ['p(95)<500'],
    'http_req_duration{type:payment}': ['p(95)<3000'],
    'http_req_failed': ['rate<0.01'],
    'iteration_duration': ['p(95)<5000'],
  },
};

const STAGING_URL = __ENV.STAGING_URL || 'https://staging.sorcyn.com';

const users = new SharedArray('users', () => JSON.parse(open('../seed-users.json')));

export default function () {
  const u = users[Math.floor(Math.random() * users.length)];
  const auth = { headers: { Authorization: `Bearer ${u.token}` } };
  const apiTag = { tags: { type: 'api' }, ...auth };
  const searchTag = { tags: { type: 'search' }, ...auth };
  const paymentTag = { tags: { type: 'payment' }, ...auth };

  // 1. Fetch posts feed (search-tagged)
  let r = http.get(`${STAGING_URL}/api/v1/posts?limit=20`, searchTag);
  check(r, { 'feed 200': (res) => res.status === 200 });
  sleep(1);

  // 2. Create post (api-tagged)
  r = http.post(`${STAGING_URL}/api/v1/posts`, JSON.stringify({
    title: `loadtest_post_${Date.now()}_${u.id}`,
    description: 'lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor.',
    categoryId: u.categoryId,
    transactionType: 'service',
    budgetMin: 50, budgetMax: 200, urgency: 'normal', durationHours: 168,
  }), { ...apiTag, headers: { ...apiTag.headers, 'Content-Type': 'application/json' } });
  check(r, { 'post 201': (res) => res.status === 201 });
  const postId = r.json('data.id');
  sleep(1);

  // 3. Submit offer (api-tagged) — only if user is a seller
  if (u.role === 'seller' && postId) {
    r = http.post(`${STAGING_URL}/api/v1/offers`, JSON.stringify({
      postId,
      price: 100,
      timeline: { canStart: '2026-05-01', completion: '2026-05-15' },
      message: 'loadtest_offer_message lorem ipsum.',
    }), { ...apiTag, headers: { ...apiTag.headers, 'Content-Type': 'application/json' } });
    check(r, { 'offer 201': (res) => res.status === 201 });
    sleep(1);
  }

  // 4. Payment intent simulation (payment-tagged) — read-only stat call to /payments/me
  r = http.get(`${STAGING_URL}/api/v1/payments/me`, paymentTag);
  check(r, { 'payments 200': (res) => res.status === 200 || res.status === 404 });

  sleep(2);
}
```

```typescript
// Run once before each load test against staging:
//   STAGING_URL=https://staging.sorcyn.com npx tsx tests/load/seed-users.ts
import { writeFileSync } from 'node:fs';

const STAGING_URL = process.env.STAGING_URL || 'https://staging.sorcyn.com';
const COUNT = 100;

interface SeedUser { id: string; token: string; role: 'buyer' | 'seller'; categoryId: string; }

async function register(role: 'buyer' | 'seller', n: number): Promise<SeedUser> {
  const email = `loadtest_${role}_${n}_${Date.now()}@sorcyn.test`;
  const res = await fetch(`${STAGING_URL}/api/v1/auth/register`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      email, password: 'LoadTestPass!2026',
      firstName: `loadtest_${role}_${n}`, lastName: 'LoadTest',
      accountType: role === 'seller' ? 'seller' : 'buyer',
      city: 'Dallas', state: 'TX', zip: '75201',
    }),
  });
  if (!res.ok) throw new Error(`register failed: ${res.status} ${await res.text()}`);
  const body = await res.json();
  // Pick the first active service category for posts
  const cats = await fetch(`${STAGING_URL}/api/v1/categories`).then(r => r.json());
  const categoryId = cats.data?.[0]?.id ?? '';
  return { id: body.data.user.id, token: body.data.tokens.accessToken, role, categoryId };
}

(async () => {
  const buyers = await Promise.all(Array.from({ length: COUNT }, (_, i) => register('buyer', i)));
  const sellers = await Promise.all(Array.from({ length: COUNT }, (_, i) => register('seller', i)));
  writeFileSync('tests/load/seed-users.json', JSON.stringify([...buyers, ...sellers], null, 2));
  console.log(`Seeded ${buyers.length + sellers.length} users → tests/load/seed-users.json`);
})().catch(e => { console.error(e); process.exit(1); });
```
