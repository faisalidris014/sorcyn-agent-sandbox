```bash
#!/usr/bin/env bash
# Phase 4 D-08 quarterly chaos drill — staging only. RESEARCH §7.
# Captures Slack screenshot evidence for #sorcyn-prod-alerts.
set -euo pipefail

# ── Staging-only guard (T-S-05) ─────────────────────────────────────────────
: "${STAGING_VPS:?Set STAGING_VPS to your staging SSH host}"
: "${STAGING_URL:?Set STAGING_URL (e.g. https://staging.sorcyn.com)}"
: "${TEST_FORCE_TOKEN:?Set TEST_FORCE_TOKEN to staging /__test/force-500 token}"
: "${STAGING_DATABASE_URL:?Set STAGING_DATABASE_URL}"

PROD_HOST_FRAGMENTS="${PROD_HOST_FRAGMENTS:-api.sorcyn.com sorcyn.com prod}"
for frag in $PROD_HOST_FRAGMENTS; do
  if [[ "$STAGING_URL" == *"$frag"* ]] && [[ "$STAGING_URL" != *"staging"* ]]; then
    echo "ABORT: STAGING_URL ($STAGING_URL) looks like production. Refusing to run." >&2
    exit 2
  fi
done

# ── Restoration trap ─────────────────────────────────────────────────────────
restore() {
  echo
  echo "[$(date -Iseconds)] === Restoring staging to healthy state ==="
  ssh "$STAGING_VPS" 'cd /opt/reverse-marketplace && docker compose up -d api' || true
  ssh "$STAGING_VPS" 'cd /opt/reverse-marketplace && docker compose restart api' || true
  echo "[$(date -Iseconds)] Restoration attempted. Verify staging /health returns 200."
}
trap restore EXIT

echo "[$(date -Iseconds)] === Step 1: Container down (uptime alert) ==="
ssh "$STAGING_VPS" 'cd /opt/reverse-marketplace && docker compose stop api'
sleep 90
echo "VERIFY: ContainerDown alert fired in #sorcyn-prod-alerts. Capture screenshot. Press enter to continue."
read -r _
ssh "$STAGING_VPS" 'cd /opt/reverse-marketplace && docker compose start api'

echo "[$(date -Iseconds)] === Step 2: 5xx burst (ApiHigh5xxBurst) ==="
for i in $(seq 1 50); do
  curl -s -o /dev/null -X GET \
    -H "X-Test-Token: $TEST_FORCE_TOKEN" \
    "$STAGING_URL/api/v1/__test/force-500" &
done
wait
sleep 90
echo "VERIFY: ApiHigh5xxBurst fired. Screenshot. Press enter."
read -r _

echo "[$(date -Iseconds)] === Step 3: Slow query (ApiP95LatencyBreach) ==="
ssh "$STAGING_VPS" "docker exec api psql \"$STAGING_DATABASE_URL\" -c 'SELECT pg_sleep(2);'" &
sleep 300
echo "VERIFY: ApiP95LatencyBreach fired. Screenshot. Press enter."
read -r _

echo "[$(date -Iseconds)] === Step 4: Stripe webhook secret revoke (StripeWebhookFailureSpike) ==="
ssh "$STAGING_VPS" 'cd /opt/reverse-marketplace && docker compose exec -T api sh -c "STRIPE_WEBHOOK_SECRET=invalid_secret_$(date +%s) docker compose up -d api"' || true
echo "Now trigger a webhook from Stripe Dashboard test mode → resend last event."
echo "Press enter when the webhook has been resent."
read -r _
sleep 180
echo "VERIFY: StripeWebhookFailureSpike fired. Screenshot. Press enter."
read -r _

echo "[$(date -Iseconds)] === Drill complete. Trap will restore staging on exit. ==="
```
